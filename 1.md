## 淘宝获取规格和价格(控制台排序和剪切板可直接粘贴到excel)
```
(() => {
    // 1. 获取全局上下文数据
    const context = window.__ICE_APP_CONTEXT__;
    if (!context) {
        console.error("未找到 window.__ICE_APP_CONTEXT__。请确认页面已加载完毕。");
        return;
    }

    const res = context.loaderData?.home?.data?.res;
    if (!res) {
        console.error("未找到数据源 res 节点。");
        return;
    }

    const skuBase = res.skuBase;
    const skuCore = res.skuCore;
    if (!skuBase || !skuCore) {
        console.error("缺少 skuBase 或 skuCore 数据。");
        return;
    }

    // 2. 建立 vid -> 规格名称的映射
    const vidToName = {};
    skuBase.props?.forEach(prop => {
        prop.values?.forEach(val => {
            vidToName[val.vid] = val.name;
        });
    });

    // 3. 建立 skuId -> 价格的映射
    const skuIdToPrice = {};
    const sku2info = skuCore.sku2info || {};
    Object.keys(sku2info).forEach(skuId => {
        const priceText = sku2info[skuId]?.price?.priceText;
        skuIdToPrice[skuId] = priceText || "无";
    });

    // 4. 组合规格与价格
    const results = skuBase.skus?.map(sku => {
        const propPath = sku.propPath || "";
        const skuId = sku.skuId;

        const vids = propPath.split(";").filter(p => p.includes(":")).map(p => p.split(":")[1]);
        const names = vids.map(vid => vidToName[vid] || vid);
        const specName = names.join(" / ");

        return {
            skuId: skuId,
            name: specName,
            price: skuIdToPrice[skuId] || "无"
        };
    }) || [];

    if (results.length === 0) {
        console.warn("未能提取到规格数据。");
        return;
    }

    // 5. 将结果转换为 Excel 可直接识别的 TSV 格式 (制表符分隔)
    let tsvContent = "SKU ID\t规格名称\t价格(元)\n";
    results.forEach(item => {
        tsvContent += `${item.skuId}\t${item.name}\t${item.price}\n`;
    });

    // 6. 执行复制到剪贴板的操作
    const doCopy = (text) => {
        // 优先使用浏览器的 DevTools 内置 copy 函数
        if (typeof copy === 'function') {
            copy(text);
            return true;
        }
        // 备用方案：创建临时文本域进行复制
        const textarea = document.createElement("textarea");
        textarea.value = text;
        textarea.style.position = "fixed"; // 避免影响页面滚动
        document.body.appendChild(textarea);
        textarea.select();
        try {
            const successful = document.execCommand("copy");
            document.body.removeChild(textarea);
            return successful;
        } catch (err) {
            document.body.removeChild(textarea);
            return false;
        }
    };

    console.clear();
    const isCopied = doCopy(tsvContent);

    // 7. 打印提示信息
    console.log(`%c成功提取 ${results.length} 条规格数据！`, "color: #ff5000; font-weight: bold; font-size: 14px;");
    if (isCopied) {
        console.log("%c[已自动复制] 数据已存入你的剪贴板，请直接到 Excel/WPS 表格中 Ctrl+V 粘贴即可。", "color: #00A67C; font-weight: bold;");
    } else {
        console.warn("自动复制可能受浏览器安全策略限制而失败，你可以手动复制以下内容：");
        console.log(tsvContent);
    }
    
    // 同时在控制台输出表格，方便预览
    console.table(results.map(r => ({ "SKU ID": r.skuId, "规格名称": r.name, "价格(元)": r.price })));
})();
```
