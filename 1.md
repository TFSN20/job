## 淘宝获取规格和价格
```
async function exportAllSkuPricesWithAntiBot() {
    // ================= 配置参数 =================
    const CONFIG = {
        baseInterval: 2000,   // 基础等待间隔（毫秒），建议 1500 - 3000
        randomBuffer: 1000,   // 随机增加的扰动时间范围（毫秒），模拟人类点击的非规律性
        batchSize: 20         // 每隔多少个规格打印一次中间结果
    };
    // ============================================

    const sleep = ms => new Promise(r => setTimeout(r, ms));
    const getRandomDelay = () => CONFIG.baseInterval + Math.floor(Math.random() * CONFIG.randomBuffer);
    
    // 获取页面规格按钮
    const skus = Array.from(document.querySelectorAll('.valueItem--smR4pNt4'));
    if (skus.length === 0) {
        console.warn("未找到规格元素，请确认页面是否完全加载或元素类名是否正确。");
        return;
    }

    console.log(`%c[防风控启动] 检测到共 ${skus.length} 个规格。将采用随机延迟策略，每 ${CONFIG.batchSize} 个保存并打印一次。`, "color: #007bff; font-weight: bold;");
    
    // 备份用户原先的选择
    const originalSelected = skus.find(el => 
        el.className.includes('Select') || 
        el.className.includes('selected') || 
        el.getAttribute('aria-checked') === 'true'
    );

    const results = [];

    for (let i = 0; i < skus.length; i++) {
        const sku = skus[i];
        
        // 读取规格文本名称
        const titleEl = sku.querySelector('.valueItemText--T7YrR8tO');
        const title = titleEl ? (titleEl.getAttribute('title') || titleEl.innerText.trim()) : `规格 ${i + 1}`;

        // 判断是否无货
        const isDisabled = sku.getAttribute('data-disabled') === 'true' || sku.className.includes('disabled');
        if (isDisabled) {
            results.push({ "规格名称": title, "对应价格": "无货/已下架" });
            
            // 即使无货，每 20 个也触发一次打印
            checkAndPrintBatch(i);
            continue;
        }

        try {
            // 产生随机延时，降低特征识别风险
            const currentDelay = getRandomDelay();
            console.log(`[进度 ${i + 1}/${skus.length}] 即将点击: "${title}"，随机等待 ${(currentDelay / 1000).toFixed(2)} 秒...`);
            await sleep(currentDelay);

            // 滚动并点击
            sku.scrollIntoView({ block: 'center', behavior: 'instant' });
            sku.click();

            // 点击后给页面留一点渲染时间
            await sleep(400);

            // 读取价格
            const priceTextEl = document.querySelector('.highlightPrice--LlVWiXXs .text--LP7Wf49z') 
                                || document.querySelector('[class*="highlightPrice"] [class*="text"]');
            const price = priceTextEl ? priceTextEl.innerText.trim() : "未抓取到价格";

            results.push({ "规格名称": title, "对应价格": `￥${price}` });
        } catch (err) {
            results.push({ "规格名称": title, "对应价格": `点击出错: ${err.message}` });
        }

        // 判断是否达到分批打印的节点
        checkAndPrintBatch(i);
    }

    // 恢复运行前的选择
    if (originalSelected) {
        originalSelected.scrollIntoView({ block: 'center', behavior: 'instant' });
        originalSelected.click();
    }

    // 最终汇总打印
    console.log("%c\n=========================================\n🎉 数据采集完毕！以下为全量数据汇总：\n=========================================", "color: #28a745; font-weight: bold;");
    console.table(results);
    return results;

    // 分批打印辅助函数
    function checkAndPrintBatch(currentIndex) {
        const itemNumber = currentIndex + 1;
        if (itemNumber % CONFIG.batchSize === 0) {
            const batchNum = itemNumber / CONFIG.batchSize;
            console.log(`%c\n--- [第 ${batchNum} 批次汇报] 已成功采集前 ${itemNumber} 个规格，本批次数据： ---`, "color: #fd7e14; font-weight: bold;");
            // 仅打印最近的这一批 20 个数据
            console.table(results.slice(itemNumber - CONFIG.batchSize, itemNumber));
        }
    }
}

// 启动提取
exportAllSkuPricesWithAntiBot();
```
