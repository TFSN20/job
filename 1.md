## 淘宝获取规格和价格
```
(() => {
    // 1. 获取全局上下文数据
    const context = window.__ICE_APP_CONTEXT__;
    if (!context) {
        console.error("未找到 window.__ICE_APP_CONTEXT__。请确认页面已加载完毕，且当前处于淘宝/天猫商品详情页。");
        return;
    }

    const res = context.loaderData?.home?.data?.res;
    if (!res) {
        console.error("未找到数据源 res 节点，可能页面数据结构已发生变动。");
        return;
    }

    const skuBase = res.skuBase;
    const skuCore = res.skuCore;
    if (!skuBase || !skuCore) {
        console.error("缺少 skuBase 或 skuCore 数据。");
        return;
    }

    // 2. 提取并建立 vid -> 规格名称的映射
    const vidToName = {};
    skuBase.props?.forEach(prop => {
        prop.values?.forEach(val => {
            vidToName[val.vid] = val.name;
        });
    });

    // 3. 提取并建立 skuId -> 价格的映射
    const skuIdToPrice = {};
    const sku2info = skuCore.sku2info || {};
    Object.keys(sku2info).forEach(skuId => {
        const priceText = sku2info[skuId]?.price?.priceText;
        skuIdToPrice[skuId] = priceText ? `￥${priceText}` : "无价格";
    });

    // 4. 组合规格与价格
    const results = skuBase.skus?.map(sku => {
        const propPath = sku.propPath || ""; // 例如 "1627207:39944300187"
        const skuId = sku.skuId;

        // 解析并组合多个维度的规格（如有颜色、尺码等多维度）
        const vids = propPath.split(";").filter(p => p.includes(":")).map(p => p.split(":")[1]);
        const names = vids.map(vid => vidToName[vid] || vid);
        const specName = names.join(" / ");

        return {
            "SKU ID": skuId,
            "规格名称": specName,
            "对应价格": skuIdToPrice[skuId] || "无"
        };
    }) || [];

    // 5. 在控制台中以表格形式直观打印
    if (results.length > 0) {
        console.clear();
        console.log(`%c成功提取到 ${results.length} 个规格数据：`, "color: #ff5000; font-weight: bold; font-size: 14px;");
        console.table(results);
    } else {
        console.warn("未能匹配到有效的规格与价格组合。");
    }
})();
```
